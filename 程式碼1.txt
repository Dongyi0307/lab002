void setup() 
{ 
for (int i = 7; i < 15; i++)
{ 
pinMode(i, OUTPUT);
} 
void loop() 
{ 
for(int j=0;j<2;j++)
{ 
 digitalWrite(i, HIGH);
 delay(50);
 digitalWrite(i, LOW); 
 delay(50); 
} 
for (int i = 14; i >6; i--) 
{ 
 digitalWrite(i, HIGH);
 delay(50);
 digitalWrite(i, LOW);
 delay(50);
}
}
